# Critério técnico → código que você lê, audita e ajusta

def calcular_escavacao(volume_m3, tipo_solo, equipamento):
    produtividade = {
        "argiloso": {"escavadeira_20t": 45, "retro": 18},
        "rochoso":  {"escavadeira_20t": 12, "retro":  5},
    }
    horas = volume_m3 / produtividade[tipo_solo][equipamento]
    return round(horas, 2)

# Resultado auditável. Você sabe exatamente o que aconteceu.
